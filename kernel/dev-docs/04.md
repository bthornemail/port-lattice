```haskell
{-# LANGUAGE DeriveGeneric #-}
{-# LANGUAGE OverloadedStrings #-}

-- | LatticeCoxeter.hs
-- |
-- | Pure Coxeter/chain-complex extraction from FIFO-first Boards.
-- | Kernel-safe: NO IO. This is your “blast radius” + geometric validation layer.
-- |
-- | Model:
-- |   0-cells: Ports ∪ Processes  (bipartite vertices)
-- |   1-cells: Incidence edges (Port ↔ Process), signed by role:
-- |       waitsOn  = -1
-- |       firesTo  = +1
-- |
-- | Boundary operator ∂₁ : C₁ → C₀ is represented as the signed incidence list
-- | (a sparse matrix in disguise).

module LatticeCoxeter
  ( Cell0(..)
  , Cell1(..)
  , Sign(..)
  , BoundaryOp(..)
  , CoxeterComplex(..)

  , extractComplex
  , boundary1

    -- * Geometry / safety checks
  , eulerCharacteristic
  , connectedComponents
  , blastRadiusFromPort
  , blastRadiusFromProc
  ) where

import           Data.Aeson (ToJSON, FromJSON)
import           Data.List (foldl')
import           Data.Map.Strict (Map)
import qualified Data.Map.Strict as Map
import           Data.Set (Set)
import qualified Data.Set as Set
import           Data.Text (Text)
import qualified Data.Text as T
import           GHC.Generics (Generic)

import           LatticeKernel
  ( Board(..), PortName(..), ProcName(..), TransportName(..)
  , Port(..), Process(..), Transport(..)
  )

--------------------------------------------------------------------------------
-- Cells and boundary operator
--------------------------------------------------------------------------------

-- | 0-cells are the bipartite vertices (ports + processes).
data Cell0
  = C0Port PortName
  | C0Proc ProcName
  deriving (Eq, Ord, Show, Generic)

instance ToJSON Cell0
instance FromJSON Cell0

-- | Whether an incidence is an input edge (-1) or output edge (+1).
data Sign = Neg | Pos
  deriving (Eq, Ord, Show, Generic)

instance ToJSON Sign
instance FromJSON Sign

signToInt :: Sign -> Int
signToInt Neg = -1
signToInt Pos =  1

-- | 1-cells are typed incidence edges in the bipartite graph.
-- | We keep them explicit so a resolver/visualizer can label them.
data Cell1 = Cell1
  { c1Id   :: Text
  , portV  :: PortName
  , procV  :: ProcName
  , sign   :: Sign     -- waitsOn = Neg, firesTo = Pos
  } deriving (Eq, Ord, Show, Generic)

instance ToJSON Cell1
instance FromJSON Cell1

-- | Sparse boundary operator row: a single 1-cell mapped to its 0-cell endpoints.
-- | Think: ∂₁(e) = (+1)*proc + (sign)*port
-- | (You can choose the opposite convention; this one is convenient for checks.)
data BoundaryOp = BoundaryOp
  { edge     :: Cell1
  , dPort    :: (Cell0, Int)   -- (C0Port p, signToInt sign)
  , dProc    :: (Cell0, Int)   -- (C0Proc q, +1)
  } deriving (Eq, Show, Generic)

instance ToJSON BoundaryOp
instance FromJSON BoundaryOp

-- | Full extracted complex.
data CoxeterComplex = CoxeterComplex
  { cells0    :: Set Cell0
  , cells1    :: Set Cell1
  , boundary  :: [BoundaryOp]   -- sparse ∂₁ representation
  } deriving (Eq, Show, Generic)

instance ToJSON CoxeterComplex
instance FromJSON CoxeterComplex

--------------------------------------------------------------------------------
-- Extraction: Board -> CoxeterComplex (PURE)
--------------------------------------------------------------------------------

-- | Build a stable edge id (deterministic string).
mkEdgeId :: ProcName -> Sign -> PortName -> Text
mkEdgeId (ProcName pn) s (PortName pt) =
  pn <> case s of { Neg -> ":in:"; Pos -> ":out:" } <> pt

-- | Extract all 1-cells from processes’ waitsOn/firesTo lists.
extractEdges :: Board -> Set Cell1
extractEdges b =
  Set.fromList $ concatMap procEdges (Map.elems (processes b))
  where
    procEdges :: Process -> [Cell1]
    procEdges pr =
      let q = procName pr
          ins  = [ Cell1 (mkEdgeId q Neg p) p q Neg | p <- waitsOn pr ]
          outs = [ Cell1 (mkEdgeId q Pos p) p q Pos | p <- firesTo pr ]
      in ins ++ outs

-- | Collect all 0-cells (ports + processes).
extractVertices :: Board -> Set Cell0
extractVertices b =
  Set.fromList $
    map (C0Port . portName) (Map.elems (ports b))
    ++
    map (C0Proc . procName) (Map.elems (processes b))

-- | Sparse boundary operator list for ∂₁.
boundary1 :: Set Cell1 -> [BoundaryOp]
boundary1 es = map toOp (Set.toList es)
  where
    toOp :: Cell1 -> BoundaryOp
    toOp e =
      BoundaryOp
        { edge  = e
        , dPort = (C0Port (portV e), signToInt (sign e))
        , dProc = (C0Proc (procV e), 1)
        }

-- | Single entry point: extract the bipartite Coxeter complex.
extractComplex :: Board -> CoxeterComplex
extractComplex b =
  let v0 = extractVertices b
      e1 = extractEdges b
      bd = boundary1 e1
  in CoxeterComplex
      { cells0   = v0
      , cells1   = e1
      , boundary = bd
      }

--------------------------------------------------------------------------------
-- Geometry / “blast radius” safety utilities (PURE)
--------------------------------------------------------------------------------

-- | Euler characteristic χ = |V| - |E| for this 1-dimensional complex.
-- | (Extends later when you add 2-cells for composed procedures.)
eulerCharacteristic :: CoxeterComplex -> Int
eulerCharacteristic cx =
  Set.size (cells0 cx) - Set.size (cells1 cx)

-- | Build an undirected adjacency map from the bipartite complex.
adjacency :: CoxeterComplex -> Map Cell0 (Set Cell0)
adjacency cx =
  foldl' addEdge Map.empty (Set.toList (cells1 cx))
  where
    addEdge :: Map Cell0 (Set Cell0) -> Cell1 -> Map Cell0 (Set Cell0)
    addEdge m e =
      let a = C0Port (portV e)
          b = C0Proc (procV e)
      in insertUndir a b m

    insertUndir :: Cell0 -> Cell0 -> Map Cell0 (Set Cell0) -> Map Cell0 (Set Cell0)
    insertUndir a b m =
      let m1 = Map.insertWith Set.union a (Set.singleton b) m
      in  Map.insertWith Set.union b (Set.singleton a) m1

-- | Connected components of the bipartite graph (ports + procs).
connectedComponents :: CoxeterComplex -> [Set Cell0]
connectedComponents cx =
  go Set.empty (Set.toList (cells0 cx))
  where
    adj = adjacency cx

    go :: Set Cell0 -> [Cell0] -> [Set Cell0]
    go _    []     = []
    go seen (x:xs)
      | Set.member x seen = go seen xs
      | otherwise =
          let comp = dfs adj x
              seen' = Set.union seen comp
          in comp : go seen' xs

dfs :: Map Cell0 (Set Cell0) -> Cell0 -> Set Cell0
dfs adj start = loop Set.empty [start]
  where
    loop vis []     = vis
    loop vis (n:ns)
      | Set.member n vis = loop vis ns
      | otherwise =
          let nxt = Set.toList (Map.findWithDefault Set.empty n adj)
          in loop (Set.insert n vis) (nxt ++ ns)

-- | “Blast radius” from a port: all vertices reachable in its component.
blastRadiusFromPort :: CoxeterComplex -> PortName -> Set Cell0
blastRadiusFromPort cx p =
  dfs (adjacency cx) (C0Port p)

-- | “Blast radius” from a process: all vertices reachable in its component.
blastRadiusFromProc :: CoxeterComplex -> ProcName -> Set Cell0
blastRadiusFromProc cx q =
  dfs (adjacency cx) (C0Proc q)

--------------------------------------------------------------------------------
-- Notes on how to use this for non-destructive defaults
--------------------------------------------------------------------------------
-- 1) Before interpreting actions, compile Board -> Lattice (kernel) and extractComplex.
-- 2) If connectedComponents has more than one component, you can refuse (or warn).
-- 3) If χ is “surprising” for your expected topology class, you can refuse (or warn).
-- 4) For board diffs, compute blastRadiusFromPort for changed ports to estimate impact.
--
-- All of the above is PURE and can run before touching the filesystem.
```

### How this plugs into the kernel (pure, no IO)

* `extractComplex :: Board -> CoxeterComplex` gives you:

  * vertices = ports + processes
  * edges = signed incidence from `waitsOn`/`firesTo`
  * `boundary :: [BoundaryOp]` is your sparse ∂₁

* Safety / “blast radius” before touching POSIX:

  * `connectedComponents` → refuse disconnected boards (or warn)
  * `eulerCharacteristic` → fast sanity signal
  * `blastRadiusFromPort` / `blastRadiusFromProc` → impact of a change
