```haskell
{-# LANGUAGE DeriveGeneric #-}
{-# LANGUAGE OverloadedStrings #-}

-- | LatticeCoxeterDiff.hs
-- |
-- | Pure diff + blast-radius analysis for CoxeterComplex.
-- | Kernel-safe: NO IO.
-- |
-- | Goal:
-- |   Given old/new Board (or old/new CoxeterComplex),
-- |   compute what changed in the bipartite incidence complex,
-- |   then compute the “blast radius” (affected components / neighborhoods)
-- |   BEFORE touching POSIX (mkfifo, netcat, ssh, etc).

module LatticeCoxeterDiff
  ( ComplexDiff(..)
  , diffComplex
  , diffBoards

  , Patch(..)
  , patchFromDiff

  , BlastReport(..)
  , blastRadiusOfPatch
  ) where

import           Data.Aeson (ToJSON, FromJSON)
import           Data.Map.Strict (Map)
import qualified Data.Map.Strict as Map
import           Data.Set (Set)
import qualified Data.Set as Set
import           Data.Text (Text)
import           GHC.Generics (Generic)

import           LatticeKernel (Board(..), PortName(..), ProcName(..))
import           LatticeCoxeter
  ( Cell0(..), Cell1(..), CoxeterComplex(..)
  , extractComplex, connectedComponents
  , blastRadiusFromPort, blastRadiusFromProc
  )

--------------------------------------------------------------------------------
-- Diff model
--------------------------------------------------------------------------------

-- | Set-diff of the complex. This is the “semantic patch” at the graph level.
data ComplexDiff = ComplexDiff
  { added0   :: Set Cell0
  , removed0 :: Set Cell0
  , added1   :: Set Cell1
  , removed1 :: Set Cell1
  } deriving (Eq, Show, Generic)

instance ToJSON ComplexDiff
instance FromJSON ComplexDiff

diffComplex :: CoxeterComplex -> CoxeterComplex -> ComplexDiff
diffComplex old new =
  ComplexDiff
    { added0   = cells0 new Set.\\ cells0 old
    , removed0 = cells0 old Set.\\ cells0 new
    , added1   = cells1 new Set.\\ cells1 old
    , removed1 = cells1 old Set.\\ cells1 new
    }

-- | Convenience: diff boards by first extracting complexes (pure).
diffBoards :: Board -> Board -> ComplexDiff
diffBoards a b = diffComplex (extractComplex a) (extractComplex b)

--------------------------------------------------------------------------------
-- Patch extraction (what changed “by kind”)
--------------------------------------------------------------------------------

-- | A structured patch: which ports/procs/edges changed.
data Patch = Patch
  { touchedPorts :: Set PortName
  , touchedProcs :: Set ProcName
  , addedEdges   :: Set Cell1
  , removedEdges :: Set Cell1
  } deriving (Eq, Show, Generic)

instance ToJSON Patch
instance FromJSON Patch

-- | Reduce a raw diff into “touched” entities.
patchFromDiff :: ComplexDiff -> Patch
patchFromDiff d =
  Patch
    { touchedPorts = touchedPorts'
    , touchedProcs = touchedProcs'
    , addedEdges   = added1 d
    , removedEdges = removed1 d
    }
  where
    touchedPorts' :: Set PortName
    touchedPorts' =
      Set.unions
        [ collectPorts0 (added0 d)
        , collectPorts0 (removed0 d)
        , collectPorts1 (added1 d)
        , collectPorts1 (removed1 d)
        ]

    touchedProcs' :: Set ProcName
    touchedProcs' =
      Set.unions
        [ collectProcs0 (added0 d)
        , collectProcs0 (removed0 d)
        , collectProcs1 (added1 d)
        , collectProcs1 (removed1 d)
        ]

collectPorts0 :: Set Cell0 -> Set PortName
collectPorts0 = Set.fromList . foldr go []
  where
    go (C0Port p) acc = p : acc
    go _         acc = acc

collectProcs0 :: Set Cell0 -> Set ProcName
collectProcs0 = Set.fromList . foldr go []
  where
    go (C0Proc q) acc = q : acc
    go _         acc = acc

collectPorts1 :: Set Cell1 -> Set PortName
collectPorts1 = Set.map portV

collectProcs1 :: Set Cell1 -> Set ProcName
collectProcs1 = Set.map procV

--------------------------------------------------------------------------------
-- Blast radius analysis (pre-POSIX safety)
--------------------------------------------------------------------------------

-- | A blast report tells you which connected components are affected,
-- | plus the union “radius set” of vertices likely impacted.
data BlastReport = BlastReport
  { patch        :: Patch
  , components   :: [Set Cell0]   -- all components in the NEW complex
  , affectedComps :: [Set Cell0]  -- subset of components intersecting touched set
  , radius       :: Set Cell0     -- union of blast sets for touched ports/procs
  } deriving (Eq, Show, Generic)

instance ToJSON BlastReport
instance FromJSON BlastReport

-- | Compute blast radius from a patch, using the NEW complex topology.
-- |
-- | Interpretation:
-- |   - components: the full partition of the new graph
-- |   - affectedComps: which components contain anything the patch touched
-- |   - radius: union of vertices reachable from each touched port/proc
-- |
-- | This is your “blast radius before mkfifo”: you can refuse if
-- | affectedComps is too large, or if radius includes forbidden cells
-- | under your policy (policy can live outside the kernel).
blastRadiusOfPatch :: CoxeterComplex -> Patch -> BlastReport
blastRadiusOfPatch newCx p =
  BlastReport
    { patch         = p
    , components    = comps
    , affectedComps = affComps
    , radius        = rad
    }
  where
    comps :: [Set Cell0]
    comps = connectedComponents newCx

    touchedVerts :: Set Cell0
    touchedVerts =
      Set.union
        (Set.map C0Port (touchedPorts p))
        (Set.map C0Proc (touchedProcs p))

    affComps :: [Set Cell0]
    affComps = filter (\c -> not (Set.null (Set.intersection c touchedVerts))) comps

    radFromPorts :: Set Cell0
    radFromPorts =
      Set.unions [ blastRadiusFromPort newCx pn | pn <- Set.toList (touchedPorts p) ]

    radFromProcs :: Set Cell0
    radFromProcs =
      Set.unions [ blastRadiusFromProc newCx qn | qn <- Set.toList (touchedProcs p) ]

    rad :: Set Cell0
    rad = Set.union radFromPorts radFromProcs
```

### How to use it (pure, pre-POSIX “blast radius”)

```haskell
import LatticeKernel
import LatticeCoxeter
import LatticeCoxeterDiff

-- oldBoard, newBoard :: Board

let oldCx = extractComplex oldBoard
let newCx = extractComplex newBoard

let d  = diffComplex oldCx newCx
let p  = patchFromDiff d
let br = blastRadiusOfPatch newCx p

-- br.affectedComps  -> which connected components are touched
-- br.radius         -> all vertices in the blast set (ports+procs)
```

### What this gives you immediately

* A deterministic “what changed?” at the **graph level** (`ComplexDiff`)
* A deterministic “what does that touch?” (`Patch`)
* A deterministic “how big is the blast radius?” (`BlastReport`)
