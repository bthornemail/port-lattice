```haskell
{-# LANGUAGE DeriveGeneric #-}
{-# LANGUAGE OverloadedStrings #-}

-- | LatticeKernel.hs
-- |
-- | Minimal FIFO-first Haskell “kernel” for your decentralized network lattice.
-- | This module is intentionally small and invariant-heavy:
-- |
-- |   * Ports are ALWAYS FIFOs (structural truth)
-- |   * Transports are projections that ATTACH to existing ports
-- |   * Processes are projections over FIFOs
-- |   * Planning is PURE (deterministic); interpretation is IO (POSIX projection)
-- |
-- | You can build everything else (Python runtime, AWK, shell, trace-resolve)
-- | as projections of the plan and event stream produced here.

module LatticeKernel
  ( -- * Core domains
    PortName(..), ProcName(..), TransportName(..)
  , FifoPath(..), Command(..)
  , Protocol(..), Mode(..)
  , Port(..), Process(..), Transport(..), Board(..)

    -- * Invariants / validation
  , BoardError(..), validateBoard

    -- * Compile / plan (pure)
  , Lattice(..), compile
  , Action(..), plan

    -- * POSIX projection (IO)
  , interpretPOSIX

    -- * ULP (free abelian group on atoms)
  , Atom(..), ULP(..)
  , ulpZero, ulpSingleton, ulpAdd, ulpNeg

    -- * Minimal trace event (optional but useful)
  , Procedure(..), Health(..), EventType(..), TraceEvent(..)
  , mkTrace
  ) where

import           Control.Exception (SomeException, try)
import           Data.Foldable (for_)
import           Data.List (nub)
import           Data.Map.Strict (Map)
import qualified Data.Map.Strict as Map
import           Data.Maybe (catMaybes)
import           Data.Text (Text)
import qualified Data.Text as T
import           GHC.Generics (Generic)
import           System.Directory (doesFileExist, createDirectoryIfMissing)
import           System.Exit (ExitCode(..))
import           System.FilePath (takeDirectory)
import           System.Process (createProcess, proc, CreateProcess(..), StdStream(..), waitForProcess)
import           System.Posix.Files (getFileStatus, isNamedPipe)
import           System.Posix.Types (ProcessID)

--------------------------------------------------------------------------------
-- Core domains
--------------------------------------------------------------------------------

newtype PortName      = PortName Text deriving (Eq, Ord, Show)
newtype ProcName      = ProcName Text deriving (Eq, Ord, Show)
newtype TransportName = TransportName Text deriving (Eq, Ord, Show)

newtype FifoPath = FifoPath FilePath deriving (Eq, Ord, Show)
newtype Command  = Command Text     deriving (Eq, Ord, Show)

-- Network “sphere” is always a projection:
data Protocol = TCP | UDP | Unix | SSH | TLS deriving (Eq, Ord, Show, Generic)
data Mode     = Listen | Connect     deriving (Eq, Ord, Show, Generic)

-- FIFO-first: a Port IS a FIFO.
data Port = Port
  { portName :: PortName
  , fifoPath :: FifoPath
  } deriving (Eq, Show, Generic)

data Process = Process
  { procName :: ProcName
  , command  :: Command
  , waitsOn  :: [PortName]   -- reads from these FIFOs
  , firesTo  :: [PortName]   -- writes to these FIFOs
  } deriving (Eq, Show, Generic)

-- Transport attaches to a PortName (FIFO), never defines one.
data Transport = Transport
  { transportName :: TransportName
  , attachPort    :: PortName
  , protocol      :: Protocol
  , mode          :: Mode
  , endpoint      :: Text         -- e.g. "tcp://:8080", "udp://host:53"
  , impl          :: Text         -- e.g. "lattice-netcat" | "socat" | "ssh"
  , args          :: [Text]       -- argv fragments (implementation-level, but still a projection)
  } deriving (Eq, Show, Generic)

data Board = Board
  { ports      :: Map PortName Port
  , processes  :: Map ProcName Process
  , transports :: Map TransportName Transport
  } deriving (Eq, Show, Generic)

--------------------------------------------------------------------------------
-- Validation (the core invariant)
--------------------------------------------------------------------------------

data BoardError
  = MissingPort PortName
  | DuplicatePortPath FifoPath
  | ProcRefsMissingPort ProcName PortName
  | TransportRefsMissingPort TransportName PortName
  deriving (Eq, Show)

validateBoard :: Board -> [BoardError]
validateBoard b =
  fifoUniqueness b ++ procPortsExist b ++ transportPortsExist b
  where
    fifoUniqueness :: Board -> [BoardError]
    fifoUniqueness Board{ports=pmap} =
      let paths = fmap fifoPath (Map.elems pmap)
          dups  = duplicates paths
      in map DuplicatePortPath dups

    procPortsExist :: Board -> [BoardError]
    procPortsExist Board{ports=pmap, processes=procs} =
      [ ProcRefsMissingPort pn p
      | (pn,pr) <- Map.toList procs
      , p <- waitsOn pr ++ firesTo pr
      , Map.notMember p pmap
      ]

    transportPortsExist :: Board -> [BoardError]
    transportPortsExist Board{ports=pmap, transports=ts} =
      [ TransportRefsMissingPort tn (attachPort t)
      | (tn,t) <- Map.toList ts
      , Map.notMember (attachPort t) pmap
      ]

duplicates :: (Ord a) => [a] -> [a]
duplicates xs =
  [ x | x <- nub xs, count x xs > 1 ]
  where
    count y = length . filter (== y)

--------------------------------------------------------------------------------
-- Compile (pure): Board -> Lattice (compiled structure)
--------------------------------------------------------------------------------

-- You can extend this with derived graphs / Coxeter export later, purely.
data Lattice = Lattice
  { latticePorts      :: [Port]
  , latticeProcesses  :: [Process]
  , latticeTransports :: [Transport]
  } deriving (Eq, Show)

compile :: Board -> Either [BoardError] Lattice
compile b =
  case validateBoard b of
    []   -> Right Lattice
              { latticePorts      = Map.elems (ports b)
              , latticeProcesses  = Map.elems (processes b)
              , latticeTransports = Map.elems (transports b)
              }
    errs -> Left errs

--------------------------------------------------------------------------------
-- Plan (pure): Lattice -> deterministic Actions (“reconcile word”)
--------------------------------------------------------------------------------

data Action
  = EnsureFifo PortName FifoPath
  | EnsureTransport Transport
  | EnsureProcess Process
  deriving (Eq, Show)

-- Deterministic ordering:
--  1) Ports (FIFOs)  2) Transports (projections)  3) Processes (projections)
plan :: Lattice -> [Action]
plan Lattice{latticePorts=ps, latticeTransports=ts, latticeProcesses=cs} =
  fifoActions ++ transportActions ++ processActions
  where
    fifoActions      = [ EnsureFifo (portName p) (fifoPath p) | p <- sortPorts ps ]
    transportActions = [ EnsureTransport t | t <- sortTransports ts ]
    processActions   = [ EnsureProcess  c | c <- sortProcesses cs ]

-- Simple stable sort keys to ensure determinism:
sortPorts :: [Port] -> [Port]
sortPorts = Map.elems . Map.fromList . fmap (\p -> (portName p, p))

sortTransports :: [Transport] -> [Transport]
sortTransports = Map.elems . Map.fromList . fmap (\t -> (transportName t, t))

sortProcesses :: [Process] -> [Process]
sortProcesses = Map.elems . Map.fromList . fmap (\p -> (procName p, p))

--------------------------------------------------------------------------------
-- POSIX projection (IO): interpret Actions
--------------------------------------------------------------------------------

-- NOTE: This interpreter is deliberately minimal and “safe by default”:
-- It creates FIFOs and starts projection processes, but does not attempt
-- complex supervision. That lives in higher layers.
interpretPOSIX :: [Action] -> IO ()
interpretPOSIX acts = for_ acts $ \a -> case a of
  EnsureFifo _ (FifoPath path) -> ensureNamedPipe path
  EnsureTransport t            -> ensureTransportProcess t
  EnsureProcess  p             -> ensureProcessProcess p

ensureNamedPipe :: FilePath -> IO ()
ensureNamedPipe path = do
  createDirectoryIfMissing True (takeDirectory path)
  exists <- doesFileExist path
  if exists
    then do
      st <- getFileStatus path
      if isNamedPipe st
        then pure ()
        else error $ "Path exists but is not a FIFO: " <> path
    else do
      -- Use /usr/bin/mkfifo if present; if not, fail loudly.
      -- (Keeps semantics aligned with POSIX tooling.)
      let cmd = "mkfifo"
      (_, _, _, ph) <- createProcess (proc cmd [path])
      ec <- waitForProcess ph
      case ec of
        ExitSuccess   -> pure ()
        ExitFailure n -> error $ "mkfifo failed (" <> show n <> "): " <> path

ensureTransportProcess :: Transport -> IO ()
ensureTransportProcess t = do
  -- Very small “start once” behavior:
  -- In a full runtime, you’d track PIDs and health-check and heal.
  --
  -- Here we just start the projection.
  let exe  = T.unpack (impl t)
      argv = map T.unpack (args t)
  _pid <- spawnDetached exe argv
  pure ()

ensureProcessProcess :: Process -> IO ()
ensureProcessProcess p = do
  -- Minimal: run via /bin/sh -c for command string.
  -- A stricter version would parse argv and avoid shell.
  let cmd = case command p of Command c -> T.unpack c
  _pid <- spawnDetached "/bin/sh" ["-c", cmd]
  pure ()

spawnDetached :: FilePath -> [String] -> IO ProcessID
spawnDetached exe argv = do
  let cp = (proc exe argv) { std_in = Inherit, std_out = Inherit, std_err = Inherit }
  e <- try (createProcess cp) :: IO (Either SomeException (Maybe a, Maybe b, Maybe c, System.Process.ProcessHandle))
  case e of
    Left err -> error $ "spawn failed: " <> show err <> " exe=" <> exe
    Right (_, _, _, ph) -> do
      -- We can’t get a PID portably from ProcessHandle without extra libs;
      -- higher layers will manage PID files if needed.
      -- Return 0 as “unknown PID” placeholder in kernel.
      _ <- pure ph
      pure 0

--------------------------------------------------------------------------------
-- ULP: free abelian group on atoms
--------------------------------------------------------------------------------

newtype Atom = Atom Text deriving (Eq, Ord, Show)

newtype ULP = ULP (Map Atom Int) deriving (Eq, Show)

ulpZero :: ULP
ulpZero = ULP Map.empty

ulpSingleton :: Atom -> Int -> ULP
ulpSingleton a k = ULP (Map.singleton a k)

ulpAdd :: ULP -> ULP -> ULP
ulpAdd (ULP a) (ULP b) =
  ULP (Map.filter (/= 0) (Map.unionWith (+) a b))

ulpNeg :: ULP -> ULP
ulpNeg (ULP a) = ULP (Map.map negate a)

--------------------------------------------------------------------------------
-- Minimal trace event (optional, but matches your “procedure” axis)
--------------------------------------------------------------------------------

data Procedure = Compile | Reconcile | Probe | Heal
  deriving (Eq, Ord, Show)

data Health = Healthy | Unhealthy | Unknown
  deriving (Eq, Ord, Show)

data EventType
  = PortMaterialized PortName FifoPath
  | TransportAttached TransportName PortName
  | ProcessEnsured ProcName
  | Note Text
  deriving (Eq, Show)

data TraceEvent = TraceEvent
  { procedure :: Procedure
  , eventType :: EventType
  , ulp       :: ULP
  } deriving (Eq, Show)

mkTrace :: Procedure -> EventType -> ULP -> TraceEvent
mkTrace = TraceEvent
```

### How to use this kernel (minimal)

1. Construct a `Board` where **every port has a FIFO path**.
2. `compile board` (pure) → `Lattice`
3. `plan lattice` (pure) → `[Action]`
4. `interpretPOSIX actions` (IO projection)

### What you get “for free”

* The FIFO-first rule is *not optional*: there is no way to represent a TCP port as a “port”.
* Network is only representable as a **Transport attached to a PortName**.
* The “logic cube” is enforced: ports/processes/procedures are orthogonal data types.
